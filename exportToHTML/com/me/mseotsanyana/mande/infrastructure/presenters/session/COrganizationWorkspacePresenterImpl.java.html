    @Override
    public void onSuccess(IResponseDTO<Object> response) {
        EAction operation = response.getAction();
        COrganizationModel organizationModel;
        CWorkspaceModel workspaceModel;

        switch (operation) {
            case Added_ORGANIZATION:
                organizationModel = (COrganizationModel) response.getData();
                if (this.iViewModel != null) {
                    // update the response model
                    if (!treeModels.containsKey(organizationModel.getOrganizationServerID())) {
                        CTreeModel treeModel;
                        treeModel = new CTreeModel(organizationModel.getOrganizationServerID(),
                                null, 0, organizationModel);
                        treeModels.add(organizationModel.getOrganizationServerID(), treeModel);
                        this.iViewModel.onAddTreeModel(new CResponseDTO<>(EAction.Added_ORGANIZATION,
                                treeModel));
                    }

                    this.iViewModel.hideProgress();
                }
                break;

            case Created_WORKSPACE:
                Log.i(TAG, "Created Workspace under an organization.");
//
//                workspaceModel = (CWorkspaceModel) response.getData();
//
//                if (this.iViewModel != null) {
//                    if (response.getData() instanceof CWorkspaceModel) {
//                        // update the response model
//                        CTreeModel treeModel = new CTreeModel(workspaceModel.getCompositeServerID(),
//                                workspaceModel.getOrganizationServerID(), 1, workspaceModel);
//                        treeModels.add(workspaceModel.getCompositeServerID(), treeModel);
//                        this.iViewModel.showTreeModel(new CResponseDTO<>(EAction.Added_WORKSPACE,
//                                treeModels));
//                    } else if (response.getData() instanceof String) {
//                        this.iViewModel.hideProgress();
//                    }
//                }
                break;

            case Added_WORKSPACE:

                if (this.iViewModel != null) {
                    if (response.getData() instanceof CWorkspaceModel) {
                        workspaceModel = (CWorkspaceModel) response.getData();
                        CTreeModel treeModel = new CTreeModel(workspaceModel.getCompositeServerID(),
                                workspaceModel.getOrganizationServerID(), 1, workspaceModel);
                        treeModels.add(workspaceModel.getCompositeServerID(), treeModel);

                        COrganizationModel modelObject = (COrganizationModel)
                                Objects.requireNonNull(treeModels.get(treeModel.getParentID())).
                                        getModelObject();

                        this.iViewModel.onAddTreeModel(new CResponseDTO<>(EAction.Added_WORKSPACE,
                                treeModel));
                        this.iViewModel.hideProgress();
                        Log.i(TAG, "Added Workspace under an organization."+modelObject.getWorkspaceBITS());
                    }
                }
                break;

            case Modified_ORGANIZATION:
                organizationModel = (COrganizationModel) response.getData();
                if (this.iViewModel != null) {
                    // update the response model
                    if (treeModels.containsKey(organizationModel.getOrganizationServerID())) {
                        CTreeModel treeModel;
                        treeModel = new CTreeModel(organizationModel.getOrganizationServerID(),
                                null, 0, organizationModel);
                        treeModels.replace(organizationModel.getOrganizationServerID(), treeModel);
                        this.iViewModel.onAddTreeModel(new CResponseDTO<>(EAction.Modified_ORGANIZATION,
                                treeModel));
                        this.iViewModel.hideProgress();
                    }
                    Log.d(TAG, "Modified Organization");

                }
                break;

            case Modified_WORKSPACE:
                Log.d(TAG, "Modified Workspace");
                break;

            case Deleted_ORGANIZATION:
                Log.d(TAG, "Deleted Organization");
                break;

            case Deleted_WORKSPACE:
                if (this.iViewModel != null) {
                    if (response.getData() instanceof CWorkspaceModel) {
                        workspaceModel = (CWorkspaceModel) response.getData();
                        Log.d(TAG, " 1 ------------- " + treeModels.keySet());
                        this.treeModels.removeItemByKey(workspaceModel.getCompositeServerID());
                        Log.d(TAG, " 2 ------------- " + treeModels.keySet());
                        this.iViewModel.onUpdateTreeModels(new CResponseDTO<>(EAction.Deleted_WORKSPACE,
                                workspaceModel.getCompositeServerID()));
                    } else if (response.getData() instanceof String) {
                        this.iViewModel.showMessage((String) response.getData());

                    }
                    this.iViewModel.hideProgress();
                }

                Log.d(TAG, "Deleted Workspace");
                break;

            case Switched_WORKSPACE:
                if (this.iViewModel != null) {
                    this.iViewModel.switchResponse((String) response.getData());
                    this.iViewModel.hideProgress();
                }
                break;
        }
    }

    @Override
    public void onError(Throwable throwable) {
        if (this.iViewModel != null) {
            this.iViewModel.showMessage(throwable.getMessage());
            this.iViewModel.hideProgress();
        }
        //Log.i(TAG, throwable.getMessage());
    }
